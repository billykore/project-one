> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Create `GET /feeds` endpoint returning cursor-paginated posts from followed users + self.

**Architecture:** New `FeedUseCase` + `FeedRepository` port + GORM adapter, following existing Clean Architecture. A lightweight `GetFollowedUsernames` method added to `FollowRepository` to avoid expensive JOINs. Cursor encoding via base64-encoded JSON of `{created_at, id}`.

**Tech Stack:** Go 1.26+, Echo v4, GORM, PostgreSQL, test using GoMock + Testify, validator v10.

## Global Constraints

- Follow existing Clean Architecture layers: Domain → Ports → UseCase → Adapters (repos) → API (handlers/DTOs)
- Go module path: `github.com/billykore/project-one`
- Cursor-based pagination using `(created_at, id)` composite, base64 JSON encoded
- Response envelope: `{"data": [...], "next_cursor": "...", "has_more": true}`
- No nested comments in feed response
- Auth required on `GET /feeds`
- Limit clamped to [1, 50], default 10
- Nil-panic-on-nil-deps constructor pattern (ponytail convention)

```go
package ports

import (
	"context"
	"encoding/base64"
	"encoding/json"
	"fmt"
	"time"

	"github.com/billykore/project-one/internal/core/domain"
)

// Cursor represents an opaque pagination cursor for feed queries.
// It encodes the (created_at, id) of the last item on the current page.
type Cursor struct {
	CreatedAt time.Time `json:"c"`
	ID        int       `json:"i"`
}

// Encode returns the base64-encoded JSON representation of the cursor.
func (c Cursor) Encode() string {
	data, _ := json.Marshal(c)
	return base64.URLEncoding.EncodeToString(data)
}

// DecodeCursor decodes a base64-encoded cursor string.
// Returns an error if the string is malformed.
func DecodeCursor(encoded string) (Cursor, error) {
	data, err := base64.URLEncoding.DecodeString(encoded)
	if err != nil {
		return Cursor{}, fmt.Errorf("invalid cursor: %w", err)
	}
	var c Cursor
	if err := json.Unmarshal(data, &c); err != nil {
		return Cursor{}, fmt.Errorf("invalid cursor: %w", err)
	}
	return c, nil
}

// FeedResult holds the result of a feed query.
type FeedResult struct {
	Posts      []*domain.Post
	NextCursor *Cursor
	HasMore    bool
}

// FeedRepository is a driven port for feed data access.
type FeedRepository interface {
	// GetFeed returns paginated posts from the given usernames using cursor-based pagination.
	// limit should be one more than requested so the caller can determine has_more.
	GetFeed(ctx context.Context, usernames []string, cursor *Cursor, limit int) ([]*domain.Post, error)
}

// FeedUseCase is a driving port for feed-related application logic.
type FeedUseCase interface {
	// GetFeed returns the feed for an authenticated user.
	GetFeed(ctx context.Context, username string, cursor *Cursor, limit int) (*FeedResult, error)
}
```

```go
package repository

import (
	"context"
	"fmt"

	"github.com/billykore/project-one/internal/core/domain"
	"github.com/billykore/project-one/internal/core/ports"
	"gorm.io/gorm"
)

type feedRepository struct {
	db *gorm.DB
}

// NewFeedRepository creates a new instance of FeedRepository.
func NewFeedRepository(db *gorm.DB) ports.FeedRepository {
	return &feedRepository{db: db}
}

func (r *feedRepository) GetFeed(ctx context.Context, usernames []string, cursor *ports.Cursor, limit int) ([]*domain.Post, error) {
	if len(usernames) == 0 {
		return []*domain.Post{}, nil
	}

	// Build query with raw SQL for composite cursor support.
	query := r.db.WithContext(ctx).
		Table("posts").
		Where("username IN ?", usernames).
		Where("deleted_at IS NULL")

	if cursor != nil {
		query = query.Where("(created_at, id) < (?, ?)", cursor.CreatedAt, cursor.ID)
	}

	var models []postModel
	if err := query.
		Order("created_at DESC, id DESC").
		Limit(limit).
		Find(&models).Error; err != nil {
		return nil, fmt.Errorf("get feed: %w", err)
	}

	posts := make([]*domain.Post, 0, len(models))
	for _, m := range models {
		posts = append(posts, m.toDomain())
	}
	return posts, nil
}
```

```go
package usecase

import (
	"context"
	"fmt"

	"github.com/billykore/project-one/internal/core/domain"
	"github.com/billykore/project-one/internal/core/ports"
)

type feedUseCase struct {
	feedRepo   ports.FeedRepository
	followRepo ports.FollowRepository
	userRepo   ports.UserRepository
	log        ports.Logger
}

// NewFeedUseCase creates a new instance of FeedUseCase.
func NewFeedUseCase(
	feedRepo ports.FeedRepository,
	followRepo ports.FollowRepository,
	userRepo ports.UserRepository,
	log ports.Logger,
) ports.FeedUseCase {
	if feedRepo == nil || followRepo == nil || userRepo == nil || log == nil {
		panic("NewFeedUseCase: dependencies must not be nil")
	}
	return &feedUseCase{
		feedRepo:   feedRepo,
		followRepo: followRepo,
		userRepo:   userRepo,
		log:        log,
	}
}

func (u *feedUseCase) GetFeed(ctx context.Context, username string, cursor *ports.Cursor, limit int) (*ports.FeedResult, error) {
	// Clamp limit.
	if limit <= 0 {
		limit = 10
	}
	if limit > 50 {
		limit = 50
	}

	// Resolve user.
	user, err := u.userRepo.GetUserByUsername(ctx, username)
	if err != nil {
		return nil, fmt.Errorf("get user by username: %w", err)
	}
	if user == nil {
		return nil, domain.ErrUserNotFound
	}

	// Get followed usernames.
	followedUsernames, err := u.followRepo.GetFollowedUsernames(ctx, username)
	if err != nil {
		return nil, fmt.Errorf("get followed usernames: %w", err)
	}

	// Build username list: self + followed.
	usernames := append([]string{username}, followedUsernames...)

	// Fetch one extra to detect has_more.
	dbLimit := limit + 1
	posts, err := u.feedRepo.GetFeed(ctx, usernames, cursor, dbLimit)
	if err != nil {
		return nil, fmt.Errorf("get feed from repo: %w", err)
	}

	result := &ports.FeedResult{
		Posts:   posts,
		HasMore: false,
	}

	if len(posts) == dbLimit {
		result.HasMore = true
		result.Posts = posts[:limit]
	}

	// Build next cursor from last post.
	if len(result.Posts) > 0 {
		last := result.Posts[len(result.Posts)-1]
		result.NextCursor = &ports.Cursor{
			CreatedAt: last.CreatedAt,
			ID:        last.ID,
		}
	}

	return result, nil
}
```

```go
package dto

import (
	"time"

	"github.com/billykore/project-one/internal/core/domain"
	"github.com/billykore/project-one/internal/core/ports"
)

// FeedResponse is the envelope for the GET /feeds response.
type FeedResponse struct {
	Data       []PostItem `json:"data"`
	NextCursor string     `json:"next_cursor"`
	HasMore    bool       `json:"has_more"`
}

// PostItem is a feed post without nested comments.
type PostItem struct {
	ID        int       `json:"id"`
	Author    string    `json:"author"`
	Title     string    `json:"title"`
	Content   string    `json:"content"`
	Tags      []string  `json:"tags"`
	LikeCount int       `json:"like_count"`
	CreatedAt time.Time `json:"created_at"`
	UpdatedAt time.Time `json:"updated_at"`
}

// ToFeedResponse converts a FeedResult to the API response DTO.
func ToFeedResponse(result *ports.FeedResult) FeedResponse {
	data := make([]PostItem, 0, len(result.Posts))
	for _, p := range result.Posts {
		data = append(data, PostItem{
			ID:        p.ID,
			Author:    p.Username,
			Title:     p.Title,
			Content:   p.Content,
			Tags:      p.Tags,
			LikeCount: p.LikeCount,
			CreatedAt: p.CreatedAt,
			UpdatedAt: p.UpdatedAt,
		})
	}

	var nextCursor string
	if result.NextCursor != nil {
		nextCursor = result.NextCursor.Encode()
	}

	return FeedResponse{
		Data:       data,
		NextCursor: nextCursor,
		HasMore:    result.HasMore,
	}
}

// Ensure domain.Post is used for the mapping above.
var _ = (*domain.Post)(nil)
```

```go
package handler

import (
	"errors"
	"net/http"
	"strconv"

	"github.com/billykore/project-one/internal/api/dto"
	"github.com/billykore/project-one/internal/core/domain"
	"github.com/billykore/project-one/internal/core/ports"
	"github.com/labstack/echo/v4"
)

type FeedHandler struct {
	feedUseCase ports.FeedUseCase
	log         ports.Logger
}

// NewFeedHandler creates a new instance of FeedHandler.
func NewFeedHandler(feedUseCase ports.FeedUseCase, log ports.Logger) *FeedHandler {
	// ponytail: nil checks removed — Go panics at method call site on nil pointer
	return &FeedHandler{
		feedUseCase: feedUseCase,
		log:         log,
	}
}

// HandleGetFeed handles the GET /feeds endpoint.
//
//	@Summary		Get feed
//	@Description	Returns paginated posts from users the authenticated user follows and their own posts.
//	@Tags			feeds
//	@Produce		json
//	@Param			cursor	query		string	false	"Pagination cursor"
//	@Param			limit	query		int		false	"Items per page (1-50, default 10)"
//	@Success		200		{object}	dto.FeedResponse
//	@Failure		400		{object}	dto.ErrorResponse
//	@Failure		401		{object}	dto.ErrorResponse
//	@Failure		500		{object}	dto.ErrorResponse
//	@Security		BearerAuth
//	@Router			/feeds [get]
func (h *FeedHandler) HandleGetFeed(c echo.Context) error {
	username, ok := c.Get("username").(string)
	if !ok {
		return c.JSON(http.StatusUnauthorized, dto.ErrorResponse{Error: "Unauthorized"})
	}

	// Parse limit.
	limit := 10
	if limitStr := c.QueryParam("limit"); limitStr != "" {
		l, err := strconv.Atoi(limitStr)
		if err != nil || l < 1 || l > 50 {
			return c.JSON(http.StatusBadRequest, dto.ErrorResponse{Error: "limit must be between 1 and 50"})
		}
		limit = l
	}

	// Decode cursor.
	var cursor *ports.Cursor
	if cursorStr := c.QueryParam("cursor"); cursorStr != "" {
		decoded, err := ports.DecodeCursor(cursorStr)
		if err != nil {
			return c.JSON(http.StatusBadRequest, dto.ErrorResponse{Error: "invalid cursor"})
		}
		cursor = &decoded
	}

	result, err := h.feedUseCase.GetFeed(c.Request().Context(), username, cursor, limit)
	if err != nil {
		if errors.Is(err, domain.ErrUserNotFound) {
			return c.JSON(http.StatusNotFound, dto.ErrorResponse{Error: "User not found"})
		}
		h.log.Error(c.Request().Context(), "failed to get feed", "username", username, "error", err)
		return c.JSON(http.StatusInternalServerError, dto.ErrorResponse{Error: "Internal server error"})
	}

	return c.JSON(http.StatusOK, dto.ToFeedResponse(result))
}
```
